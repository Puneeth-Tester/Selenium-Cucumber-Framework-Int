<<<<<<<<< CMD for Multiple Browsers >>>>>>>>>>

# executes on firefox browser
mvn clean test -Dbrowser=firefox

# executes on default browser --> chrome
mvn clean test

# executes on chrome browser
mvn clean test -Dbrowser=chrome

<<<<<<<<< CMD for Multiple Environments >>>>>>>>>>

# picks up "PROD" env
mvn clean test -Denv=PROD

# picks up the default env --> "STAGE"
mvn clean test

# picks up "STAGE" env
mvn clean test -Denv=STAGE

mvn clean test

<<<<<<<<<<<<<<<< >>>>>>>>>>>>>>>>

mvn clean test -Denv=STAGE -Dbrowser=chrome -Dcucumber.filter.tags=@smoke

mvn exec:java -Dexec.classpathScope=test -Dexec.mainClass=io.cucumber.core.cli.Main -Dcucumber.filter.tags=@smoke
mvn exec:java -Dexec.classpathScope=test -Dexec.mainClass=io.cucumber.core.cli.Main -Dexec.args="--tags @smoke"
mvn exec:java -Dexec.classpathScope=test -Dexec.mainClass=io.cucumber.core.cli.Main -Dexec.args="--tags @smoke --threads 2"
